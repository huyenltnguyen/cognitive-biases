exercise:
  kind: multiple-choice
  mode: single
  question: 'A person has been learning a musical instrument for two years. She does not enjoy practicing, rarely plays outside of lessons, and recently realized she would rather spend that time and money on something else. But she has already invested two years and feels it would be a waste to stop. Which response best applies the tool?'
  options:
    - id: keep-going
      label: 'Keep going — two years of progress should not be thrown away'
      explanation: 'This is the pattern the tool is designed to counter. The two years are already spent and cannot be recovered.'
    - id: start-today
      label: 'Ask whether she would choose to begin this now if she were starting from scratch.'
      explanation: 'Correct. This strips out the sunk cost and evaluates the decision based on what is ahead.'
    - id: three-more-months
      label: 'Commit to three more months to make sure before deciding'
      explanation: 'This delays the decision but does not address the sunk cost. The same reasoning will apply in three months.'
    - id: ask-friends
      label: 'Ask friends and family whether they think she should continue'
      explanation: 'Other people may offer useful perspectives, but this does not apply the tool. The question is whether she would choose this if starting fresh.'
  correct_option_ids:
    - start-today
