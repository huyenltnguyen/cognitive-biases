exercise:
  kind: multiple-choice
  mode: single
  question: 'You are about to join a team meeting where the group will decide between two approaches for a project. You have reviewed both and have a clear preference. What should you do before the discussion begins?'
  options:
    - id: wait-and-listen
      label: 'Wait to hear what others think so you can build on their reasoning.'
      explanation: 'This is the pattern the tool is designed to counter. Hearing others first makes it harder to hold your own view.'
    - id: prepare-both-sides
      label: 'Prepare arguments for both sides so you can support whichever the group prefers.'
      explanation: 'This prepares you to go along with the group, not to contribute your own judgment.'
    - id: write-it-down
      label: 'Write down which approach you prefer and your main reason before the discussion begins.'
      explanation: 'Correct. Committing to your position privately before the group discussion protects your independent judgment.'
    - id: align-with-lead
      label: 'Talk to the team lead privately beforehand to align your positions.'
      explanation: 'This replaces group conformity with one-on-one conformity. The tool is about preserving your independent view.'
  correct_option_ids:
    - write-it-down
